Ball.prototype.draw_arrow = function (context) { context.beginPath(); var headlen = 8; var angle = Math.atan2(this.vy, this.vx); context.moveTo(this.x, this.y); context.lineTo(this.x + this.vx, this.y + this.vy); context.lineTo((this.x + this.vx)-headlen*Math.cos(angle-Math.PI/6), (this.y + this.vy)-headlen*Math.sin(angle-Math.PI/6)); context.moveTo(this.x + this.vx, this.y + this.vy); context.lineTo((this.x + this.vx)-headlen*Math.cos(angle+Math.PI/6), (this.y + this.vy)-headlen*Math.sin(angle+Math.PI/6)); context.strokeStyle = '#09A948'; random_color(); context.lineWidth = 2; context.stroke(); };
